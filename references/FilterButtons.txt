
import React from "react";
import { Button } from "@/components/ui/button";

interface FilterButtonsProps {
  filters: string[];
  activeFilter: string;
  onFilterChange: (filter: string) => void;
  className?: string;
}

export const FilterButtons: React.FC<FilterButtonsProps> = ({ 
  filters, 
  activeFilter, 
  onFilterChange,
  className = "" 
}) => {
  return (
    <div className="px-12 py-4 flex flex-wrap gap-2">
      {filters.map((filter) => (
        <button
          key={filter}
          className={`flex h-10 items-center justify-center px-6 py-2 rounded-full text-sm font-medium tracking-[0.1px] ${
            activeFilter === filter
              ? "bg-[rgba(203,203,203,1)] text-black"
              : "bg-transparent text-white hover:bg-white/10 border border-white/10"
          } ${className}`}
          onClick={() => onFilterChange(filter)}
        >
          {filter}
        </button>
      ))}
    </div>
  );
};
